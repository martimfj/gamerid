create table Users (
    id          INT(11)     not null    auto_increment,
    username    varchar(32) not null,
    password    varchar(32) not null,    
    email       varchar(64) not null,
    steam       varchar(32), 
    riot        varchar(32), 
    battlenet   varchar(32), 
    discord     varchar(32), 
    
    PRIMARY     KEY(id),
    UNIQUE      KEY(username),
    UNIQUE      KEY(email),
    UNIQUE      KEY(steam),
    UNIQUE      KEY(riot),
    UNIQUE      KEY(battlenet),
    UNIQUE      KEY(discord)
    );
